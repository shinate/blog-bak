[php]get_included_files()[/php]

[php]get_declared_classes()[/php]

[php]get_declared_interfaces()[/php]

[php]get_defined_functions()[/php]

[php]get_defined_vars()[/php]

[php]get_current_user()[/php]

[php]get_defined_constants()[/php]
